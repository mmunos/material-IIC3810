\begin{frame}
\frametitle{Motivación: coloración de grafos}

Una grafo $G = (N,A)$ se dice $k$-coloreable si existe una función $f : N \to \{1, \ldots, k\}$ tal que:
\begin{itemize}
\item Si $(a,b) \in A$, entonces $f(a) \neq f(b)$
\end{itemize}

\vs{8}

\visible<2->{
El número cromático de un grafo $G$, denotado como $\chi(G)$, se define como el menor número $k$ tal que $G$ es $k$-coloreable.
\begin{itemize}
\item Vale decir, $G$ es $k$-coloreable y no $(k-1)$-coloreable
\end{itemize}}


\end{frame}

%--------------------------------------------------
\begin{frame}
\frametitle{El problema de coloración de un grafo}

Considere los siguientes problemas relacionados con el cálculo del número cromático de un grafo:
\begin{align*}
&\alert{\CROM^\leq} \ \alert{=} \ \alert{\{ (G,k) \mid G \text{ es un grafo, }}\\
&\hspace{90pt}\alert{ k \text{ es un número natural y } \chi(G) \leq k \}}\\
&\alert{\CROM^\geq} \ \alert{=} \ \alert{\{ (G,k) \mid G \text{ es un grafo, }}\\
&\hspace{90pt}\alert{ k \text{ es un número natural y } \chi(G) \geq k \}}
\end{align*}

\vs{6}

\visible<2->{
\begin{ejercicio}
Demuestre que $\CROM^\leq$ es $\np$-completo y $\CROM^\geq$ es $\co\np$-completo.
\end{ejercicio}}


\end{frame}

\frametitle{Cálculo del número cromático}

Considere ahora el problema de decisión asociado al cálculo exacto del número cromático de un grafo:
\begin{align*}
&\alert{\CROM} \ \alert{=} \ \alert{\{ (G,k) \mid G \text{ es un grafo, }}\\
&\hspace{90pt}\alert{ k \text{ es un número natural y } \chi(G) = k \}}
\end{align*}

\vs{6}

\visible<2->{
¿Cuál es la complejidad de $\CROM$?
\begin{itemize}
\item ¿Está este problema en $\np$? ¿Está en $\co\np$? ¿Es completo para alguna de estas clases?
\end{itemize}}

\end{frame}

\begin{frame}
\frametitle{Motivación: evaluación de expresiones aritméticas}

Una expresión aritmética es definida recursivamente por las siguientes reglas:
\vs{1}
\begin{itemize}
\item Si $n \in \mathbb{N}$, entonces $n$ es una expresión aritmética

\vs{1}

\item Si $e_1$ y $e_2$ son expresiones aritméticas, entonces $(e_1 + e_2)$ y $(e_1 \cup e_2)$ son expresiones aritméticas
\end{itemize}

\vs{8}

\visible<2->{
\begin{ejemplo}
$5$, $(5 + 7)$ y $((5 + 7) \cup (4 \cup (6 + 11)))$ son expresiones aritméticas.
\end{ejemplo}}



\end{frame}

%--------------------------------------------------
\begin{frame}
\frametitle{Evaluación de expresiones aritméticas}

El lenguaje $L(e)$ definido por una expresión aritmética $e$ es definido recursivamente por las siguiente reglas:
\vs{1}
\begin{itemize}
\item Si $e = n$ con $n \in \mathbb{N}$, entonces $L(e) = \{n\}$

\vs{1}

\item Si $e = (e_1 + e_2)$, entonces $L(e) = \{ n + m \mid n \in L(e_1)$ y $m \in L(e_2)\}$

\vs{1}

\item Si $e = (e_1 \cup e_2)$, entonces $L(e) = L(e_1) \cup L(e_2)$
\end{itemize}

\end{frame}

%--------------------------------------------------
\begin{frame}
\frametitle{Evaluación de expresiones aritméticas}

\begin{ejemplo}
El lenguaje definido por algunas expresiones aritméticas:
\vs{1}
\begin{itemize}
\item $L(5) = \{5\}$ y $L(7) = \{7\}$

\vs{1}

\item Si $e = (5 + 7)$, entonces $L(e) = \{ n + m \mid n \in L(5)$ y $m \in L(7)\} = \{12\}$

\vs{1}

\item Si $e = (6 + 11)$, entonces $L(e) = \{17\}$

\vs{1}

\item Si $e = (4 \cup (6 + 11))$, entonces $L(e) = L(4) \cup L((6+11)) = \{4\} \cup \{17\} = \{4,17\}$

\vs{1}

\item Si $e = (1 \cup 3) + (7 \cup 13)$, entonces $L(e) = \{8,10,14,16\}$
\end{itemize}
\end{ejemplo}

\end{frame}

\begin{frame}
\frametitle{MT con oráculo}

\begin{definicion}
MT determinista con oráculo para $A \subseteq
\Sigma^*$: \alert{$M^A = (Q, \Sigma, \Gamma, q_0, \delta, F)$}  

\begin{itemize}
\item $Q$ es un conjunto finito de estados tal que \alert{$q_?,
q_\yes, q_\no \in Q$} 

\item $\Sigma$ es un alfabeto finito tal que $\vdash, \B \not\in
\Sigma$

\item $\Gamma$ es un alfabeto finito tal que $\Sigma \cup \{\vdash,
\B\} \subseteq \Gamma$

\item $q_0 \in Q$ es el estado inicial

\item $F \subseteq Q$ es un conjunto de estados finales

\item $\delta$ es una función parcial:
\begin{eqnarray*}
\alert{\delta} & \alert{:} & \alert{Q \times \Gamma \times \Gamma
\rightarrow Q \times \Gamma \times \{\leftarrow, \Box, \rightarrow\} \times \Gamma
\times \{\leftarrow, \Box, \rightarrow\}}  
\end{eqnarray*}
La segunda cinta es la \alert{cinta de consulta}
\end{itemize}
\end{definicion}

\end{frame}

%--------------------------------------------------
\begin{frame}
\frametitle{MT con oráculo: Funcionamiento}

\begin{block}{}
La definición anterior puede extenderse fácilmente al caso de no
determinismo.
\end{block}

\vs{6}

\visible<2->{Una MT $M$ con oráculo para $A$ funciona como una MT
tradicional excepto cuando entra al estado $q_?$:}
\vs{2}
\begin{itemize}
\visible<3->{\item Cinta de consulta: \alert{$\vdash w \B \B \cdots$},
para $w \in \Sigma^*$

\vs{2}

La cabeza lectora de la cinta de consulta está en la posición 1}

\vs{3}

\visible<4->{\item $M$ invoca al oráculo para $A$, y su siguiente
estado es \alert{$q_\yes$} o \alert{$q_\no$} dependiendo de su
respuesta
\begin{itemize}
\alert{\item $w \in A$ si y sólo si el estado es $q_\yes$}
\end{itemize}}
\end{itemize} 

\end{frame}

\begin{frame}
\frametitle{La jerarquía polinomial y $\exptime$}

{\small
¿Puede ser cierto que $\exptime \subseteq \ph$?
\begin{itemize}
\visible<2->{\item Esto implicaría que $\ph$ tiene problemas
completos}
\end{itemize}

\vs{6}

\visible<3->{¿Puede tener problemas completos $\ph$?}
\begin{itemize}
\visible<4->{\item Cada clase $\Sigma^P_k$ es cerrada bajo $\rp$
\begin{itemize}
{\footnotesize
\item Si $L_1 \rp L_2$ y $L_2 \in \Sigma^P_k$, entonces $L_1 \in \Sigma^P_k$}
\end{itemize}}

\alert{\visible<5->{\item Si $\ph$ tiene un problema completo, entonces la
jerarquía polinomial colapsa a algún nivel finito}}
\end{itemize}

\vs{6}

\visible<6->{
\begin{proposition}
Si la jerarquía polinomial no colapsa a algún nivel finito, entonces
$\ph \subsetneq \exptime$.
\end{proposition}}

}

\end{frame}

\begin{frame}
\frametitle{El colapso de la jerarquía polinomial: un resultado fundamental}

\begin{teorema}
Para $k \geq 1$:
\vs{1}
\begin{itemize}
\alert{\item[(a)] Si $\Sigma^P_k = \Pi_k^P$, entonces $\ph = \Sigma^P_k$

\vs{1}

\item[(b)] Si $\Sigma^P_k = \Delta_k^P$, entonces $\ph = \Delta^P_k$}
\end{itemize}
\end{teorema}



\end{frame}

\begin{frame}
\frametitle{Problemas completos en la jerarquía polinomial}

El lenguaje $\qbf_i$ ($i \geq 1$) está formado por todas las fórmulas
proposicionales cuantificadas que son válidas y de la forma:
\begin{align*}
\alert{\exists x_{1,1}} & \alert{\cdots \exists x_{1,m_1}}\\
\alert{\forall x_{2,1}} & \alert{\cdots \forall x_{2,m_2}}\\
\alert{\exists x_{3,1}} & \alert{\cdots \exists x_{3,m_3}}\\
&\alert{\cdots}\\
\alert{Q_i x_{i,1}} & \alert{\cdots Q_i x_{i,m_i} \, \varphi} 
\end{align*}
donde:
\begin{itemize}
\item $Q_i =\exists$ si $i$ es impar y $Q_i = \forall$ si $i$ es par

\item $\varphi$ es una fórmula proposicional sobre las variables 
$x_{1,1}$, $\ldots$, $x_{1,m_1}$, $\ldots$, $x_{i,1}$, $\ldots$,
$x_{i,m_i}$
\end{itemize}
\end{frame}

%--------------------------------------------------
\begin{frame}
\frametitle{Un problema completo para $\Sigma^P_k$}

La clase de problemas $\{\qbf_i\}_{i \geq 1}$ es adecuada para
representar la jerarquía polinomial. 

\vs{8}


\begin{teorema}
Para cada $k \geq 1$, $\qbf_k$ es $\Sigma^P_k$-completo.
\end{teorema}



\end{frame}

$\overline{\EQUIV}$ es $\Sigma^P_2$-completo.
\begin{itemize}
\item Se deduce que $\EQUIV$ es $\Pi^P_2$-completo
\end{itemize}
\end{teorema}

\vs{6}

\visible<2->{
Hay problemas naturales que son completos para los distintos niveles de la jerarquía polinomial.}

\end{frame}

\begin{frame}
\frametitle{Comentario final: Una noción de reducción más general}

Los problemas en las transparencias anteriores son completos bajo la noción de reducción $\rp$
\begin{itemize}
\item Una propiedad fundamental de esta noción de reducción: si $L_1 \rp L_2$ y $L_2 \in \ptime$, entonces $L_1 \in \ptime$
\end{itemize}

\vs{8}

\visible<2->{Podemos generalizar $\rp$ manteniendo esta propiedad fundamental:}
\begin{center}
\visible<3->{\alert{$L_1 \tur L_2$ \ si y sólo si \ $L_1 \in \ptime^{L_2}$}}
\end{center}

\end{frame}

\begin{frame}
\frametitle{Comentario final: Una noción de reducción más general}

$\tur$ es llamada \alert{reducción de Turing de tiempo polinomial}

\vs{6}

\visible<2->{
\begin{teorema}
\begin{itemize}
\item Si $L_1 \rp  L_2$, entonces $L_1 \tur L_2$

\vs{2}

\item Existen lenguajes $L_1$ y $L_2$ tales que $L_1 \tur L_2$ y $L_1 \not\rp  L_2$
\end{itemize}
\end{teorema}}

\end{frame}



%--------------------------------------------------
\begin{frame}
\frametitle{Demostración del teorema}

{\small

La primera parte del teorema es fácil de demostrar, sólo vamos a demostrar la segunda parte.
\begin{itemize}
\item Vamos a considerar los lenguajes sobre el alfabeto $\{0,1\}$
\end{itemize}

\vs{8}

\visible<2->{
Recuerde la definición de las siguientes clases de complejidad:
\vs{-1}
\alert{
\begin{eqnarray*}
\dec & = & \{ L \subseteq \{0,1\}^*  \mid L \text{ es un lenguaje decidible}\}\\
\re & = & \{ L \subseteq \{0,1\}^* \mid L \text{ es un lenguaje recursivamente enumerable}\}
\end{eqnarray*}}
}

}

\end{frame}

\begin{frame}
\frametitle{Un lenguaje que separa $\re$ de $\co\re$}

Recuerde la definición del problema de la parada de la máquina de Turing:
\vs{-1}
\begin{multline*}
U \ = \ \{ (M,w) \mid M \text{ es una MT, } w \in \{0,1\}^* \text{ y}\\ M \text{ se detiene con entrada } w \}
\end{multline*}

\vs{6}

\visible<2->{Sabemos que $U \not\in \dec$ y $U \in \re$}
\begin{itemize}
\visible<3->{\item Concluimos que $U \not\in \co\re$}

\visible<4->{\item Entonces tenemos que $\overline{U} \in \co\re$ y $\overline{U} \not\in \re$}
\end{itemize}

\end{frame}

\begin{frame}
\frametitle{Parte final de la demostración}

Sabemos que si $L \rp U$, entonces $L \in \re$
\begin{itemize}
\item ¿Cómo se demuestra esto?
\end{itemize}

\vs{8}

\visible<2->{Tenemos entonces que $\overline{U} \not\rp U$}

\vs{8}

\visible<3->{\alert{Esto concluye la demostración puesto que $\overline{U} \tur U$}}
\begin{itemize}
\visible<4->{\item De hecho, para cada lenguaje $L$ se tiene que $\overline{L} \tur L$ \qed}
\end{itemize}


\end{frame}
